PROCEDURE swap(a, b) IS
VAR c
BEGIN
    c := a;
    a := b;
    b := c;
END

PROCEDURE gcd(a, b, c) IS
VAR x, y
BEGIN
    x := a;
    y := b;
    WHILE y!=0 DO
        IF x >= y THEN
            x := x - y;
        ELSE
            swap(x, y)
        ENDIF
    ENDWHILE
    c := x;
END

PROGRAM IS
VAR a, b, x
BEGIN
    gcd(a, b, x)
END
